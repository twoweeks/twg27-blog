import { donations } from '../../data/donations';
import { formatDate } from '../../utils/date';
import { parseRoubles, formatRoubles } from '../../utils/money';

В этом посте будут собраны все пожертвования в призовой фонд конкурса. Список обновляется.

<ul>
	{donations.map(donation => (
		<li>
			{formatDate(donation.dateIso, 'short')} · {formatRoubles(donation.amountRoubles, 'long')} ·{' '}
			{donation.nickname || 'Аноним'} ·{' '}
			{donation.comment ? <q>{donation.comment}</q> : 'без комментария'}
		</li>
	))}
</ul>

Спасибо всем за [поддержку конкурса](/page/rules/#-о-призовом-фонде)!
